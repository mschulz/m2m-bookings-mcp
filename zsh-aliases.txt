gitall() {
	git status
	RESULT=$?
	if [ $RESULT -eq 0 ]; then
		echo 'Update required!'
		git add .
		if [ -n "$1" ]
		then
			git commit -m "$1"
		else
			git commit -m update
		fi
		git push origin master
	else
		echo 'All up to date!'
	fi
}
